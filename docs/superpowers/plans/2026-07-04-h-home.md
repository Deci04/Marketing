# H — Home azioni per-ruolo — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Trasformare la sezione "Da fare adesso" della home in una **lista di azioni imperative personalizzata per ruolo**, ordinata per urgenza e a rumore zero. **Luca** (collaborator) vede deadline aggregate ("⏳ Hai 4 giorni per consegnare i prossimi 4 Reel") + "✅ N montati da confermare"; **Matteo** (admin) vede "🎬 N materiali da revisionare". Tap sull'aggregato → espande la lista dei singoli contenuti.

**Architecture:** Tutta la logica di aggregazione/classificazione vive come **funzioni pure in `src/lib/workflow.ts`** (testabili senza DB), riusando `workflowState`. La home (`home/page.tsx`, Server Component async già esistente) mappa i `contents` già caricati da `listContents`, chiama `homeActions(...)` e rende gli aggregati con `<details>/<summary>` nativi (nessun client component, nessun JS custom). Ruolo derivato da `ctx.user.isAdmin` (admin = Matteo, collaborator = Luca). Nessuno schema, nessuna query nuova, nessuna migration.

**Tech Stack:** Next.js 16.2.9 (App Router, RSC), TypeScript, Tailwind, vitest 4. Nessuna dipendenza nuova.

```ts
const DAY_MS = 86_400_000;

/** Indice giorno-calendario in UTC (le deadline sono date-only a mezzanotte UTC). */
export function dayNumUTC(d: Date): number {
  return Math.floor(
    Date.UTC(d.getUTCFullYear(), d.getUTCMonth(), d.getUTCDate()) / DAY_MS
  );
}

/** Giorni-calendario UTC da `now` a `deadline`. 0 = oggi, 1 = domani, <0 = in ritardo. */
export function daysUntil(deadline: Date, now: Date): number {
  return dayNumUTC(deadline) - dayNumUTC(now);
}
```

```ts
export type HomeContent = {
  id: string;
  title: string;
  format: string | null;
  deliveredAt: Date | null;
  confirmedAt: Date | null;
  hasMontato: boolean;
  block: { lucaDeliveryAt: Date | null } | null;
};

export type DeadlineGroup = {
  daysUntil: number;
  count: number;
  noun: string;
  contentIds: string[];
};

/** Plurale imperativo del formato per il testo aggregato. */
const FORMAT_PLURAL: Record<string, string> = {
  REEL: "Reel",
  CAROUSEL: "Caroselli",
  STORY: "Storie",
  LONG_VIDEO: "Video",
};

/** Raggruppa i contenuti ancora "Da consegnare" per la scadenza di Luca
 *  (`block.lucaDeliveryAt`), tenendo solo le scadenze entro 3 giorni
 *  (quelle in ritardo incluse). Ordinati dal più urgente al meno urgente. */
export function lucaDeadlineGroups(
  contents: HomeContent[],
  now: Date
): DeadlineGroup[] {
  const buckets = new Map<number, HomeContent[]>();
  for (const c of contents) {
    if (workflowState(c) !== "Da consegnare") continue;
    const dl = c.block?.lucaDeliveryAt;
    if (!dl) continue;
    const d = daysUntil(dl, now);
    if (d > 3) continue; // solo scadenze imminenti (≤ 3 giorni)
    const bucket = buckets.get(d);
    if (bucket) bucket.push(c);
    else buckets.set(d, [c]);
  }

  return [...buckets.entries()]
    .map(([d, items]) => {
      const formats = new Set(items.map((i) => i.format));
      const first = items[0].format ?? "";
      const noun =
        formats.size === 1 && FORMAT_PLURAL[first] ? FORMAT_PLURAL[first] : "contenuti";
      return {
        daysUntil: d,
        count: items.length,
        noun,
        contentIds: items.map((i) => i.id),
      };
    })
    .sort((a, b) => a.daysUntil - b.daysUntil);
}
```

```ts
export type HomeRole = "luca" | "matteo";

export type HomeAction = {
  key: string;
  emoji: string;
  text: string;
  urgency: number; // minore = più urgente
  contentIds: string[];
};

/** Testo imperativo per una deadline aggregata di Luca. */
function deadlineText(g: DeadlineGroup): string {
  const what = `i prossimi ${g.count} ${g.noun}`;
  if (g.daysUntil < 0) return `Sei in ritardo: consegna ${what}`;
  if (g.daysUntil === 0) return `Consegna oggi ${what}`;
  if (g.daysUntil === 1) return `Hai 1 giorno per consegnare ${what}`;
  return `Hai ${g.daysUntil} giorni per consegnare ${what}`;
}

/** Azioni della home ordinate per urgenza, per ruolo. Solo azioni non vuote. */
export function homeActions(
  contents: HomeContent[],
  role: HomeRole,
  now: Date
): HomeAction[] {
  if (role === "matteo") {
    const toReview = contents.filter((c) => workflowState(c) === "Da revisionare");
    if (!toReview.length) return [];
    return [
      {
        key: "review",
        emoji: "🎬",
        text: `${toReview.length} materiali da revisionare`,
        urgency: 50,
        contentIds: toReview.map((c) => c.id),
      },
    ];
  }

  // Luca
  const actions: HomeAction[] = lucaDeadlineGroups(contents, now).map((g) => ({
    key: `deadline-${g.daysUntil}`,
    emoji: "⏳",
    text: deadlineText(g),
    urgency: g.daysUntil, // più vicino = più urgente
    contentIds: g.contentIds,
  }));

  const toConfirm = contents.filter((c) => workflowState(c) === "Da confermare");
  if (toConfirm.length) {
    actions.push({
      key: "confirm",
      emoji: "✅",
      text: `${toConfirm.length} montati da confermare`,
      urgency: 100, // dopo le deadline imminenti
      contentIds: toConfirm.map((c) => c.id),
    });
  }

  return actions.sort((a, b) => a.urgency - b.urgency);
}
```
